    <script>

        document.addEventListener('DOMContentLoaded', () => {

            const landingPage = document.getElementById('landing-page');

            const investorPage = document.getElementById('investor-page');

            const investorAccessBtn = document.getElementById('investor-access-btn');

            const backToHomeBtn = document.getElementById('back-to-home-btn');


            // Function to switch views

            function showPage(pageToShow, pageToHide) {

                if (pageToHide) {

                    pageToHide.classList.add('opacity-0');

                    // Wait for opacity transition to finish before hiding

                    setTimeout(() => {

                        pageToHide.classList.add('hidden');

                        pageToHide.classList.remove('opacity-0'); // Reset opacity for next time

                    }, 500); // Match CSS transition duration

                }

                if (pageToShow) {

                    pageToShow.classList.remove('hidden');

                    // Trigger reflow before changing opacity for transition to work

                    void pageToShow.offsetWidth;

                    pageToShow.classList.add('opacity-100');

                }

                window.scrollTo({ top: 0, behavior: 'smooth' });

            }


            if (investorAccessBtn && landingPage && investorPage) {

                investorAccessBtn.addEventListener('click', () => {

                    // Make investor page visible and start transition

                    investorPage.classList.remove('hidden');

                    void investorPage.offsetWidth; // Trigger reflow

                    investorPage.classList.remove('opacity-0'); // Remove initial opacity-0


                    // Hide landing page

                    landingPage.classList.add('opacity-0');

                     setTimeout(() => {

                        landingPage.classList.add('hidden');

                    }, 500);

                    window.scrollTo({ top: 0, behavior: 'smooth' });

                });

            }


            if (backToHomeBtn && landingPage && investorPage) {

                backToHomeBtn.addEventListener('click', () => {

                    // Make landing page visible and start transition

                    landingPage.classList.remove('hidden');

                    void landingPage.offsetWidth; // Trigger reflow

                    landingPage.classList.remove('opacity-0');


                    // Hide investor page

                    investorPage.classList.add('opacity-0');

                    setTimeout(() => {

                        investorPage.classList.add('hidden');

                    }, 500);

                    window.scrollTo({ top: 0, behavior: 'smooth' });

                });

            }


            // Initial page visibility setup

            if (landingPage) {

                 landingPage.classList.remove('hidden', 'opacity-0'); // Ensure landing page is visible

            }

            if (investorPage) {

                investorPage.classList.add('hidden', 'opacity-0'); // Ensure investor page is hidden and transparent initially

            }

        });

    </script>
